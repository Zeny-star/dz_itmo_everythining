\qs{}{
	Посчитать производные следующих функций
	\begin{enumerate}
		\item $\arctg{x}$
		\item $\dfrac{1}{1+x^2\sin^2{x}}$
		\item $\dfrac{\ln{x}}{\sin{x}}$
		\item $\ln{(\ln{x})}$
		\item $\exp^{x+\cos{x}}$
	\end{enumerate}
}
\sol
\begin{enumerate}
	\item $\arctg'{x} = \dfrac{1}{\tg'{\arctg{x}}}\Rightarrow \tg'{\arctg{x}}=\dfrac{1}{\cos^2{\arctg{x}}}$
\end{enumerate}
