\qs{}{
Выяснить, какими свойствами обладает операция $*$ на множестве $M$, если:
\begin{enumerate}
	\item $(M, \,\cdot)$, где $M=\mathbb{N},\; \mathbb{Z},\; \mathbb{Q},\; \mathbb{R}$
	\item $(M^*, \,\cdot)$, где $M=\mathbb{N},\; \mathbb{Z},\; \mathbb{Q},\; \mathbb{R}\;$  $M^*$ --- ненулевые элементы этого множества
	\item $M=\mathbb{N},\; x*y=\text{НОД}(x,\; y)$
	\item $M=\mathbb{Z}, \; x*y=x^2+y^2$
\end{enumerate}
